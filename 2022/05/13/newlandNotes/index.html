DB32_SUB_SYS_REFUND

SELECT

*

– o.CREATE_DATE,

– o.REFUND_DATE,

– o.PHONE_NUMBER,

– o.DEST_NUMBER,

– o.CITY_ID,

– o.DEST_CITY_ID,

– c.REFUND_OP_ID,

– inf.REFUND_LEVEL,

– o.AREA_ID,

– inf.GROUP_ID,

– o.REFUND_REMARK,

– o.ORDER_ID,

– o.SEQ_REL,

– o.REFUND_BUSI_TYPE,

– b.audit_time,

– b.audit_op_id,

– infe.REFUND_LEVEL,

– infe.AREA_ID,

– c.CHECK_ID,

– c.CHECK_TYPE,

– c.CHECK_TIME

FROM REFUND_CHECK c LEFT JOIN REFUND_ORDER o on c.ORDER_ID = o.ORDER_ID

left join refund_op_info inf on c.REFUND_OP_ID = inf.OP_ID

left join (

SELECT a.order_id as ORDER_ID,max(a.AUDIT_SEQ) as AUDIT_SEQ FROM refund_order_trans_inf a

where a.audit_process = 2 and a.audit_status = 2 GROUP BY a.ORDER_ID

) b on b.order_id == o.ORDER_ID

left JOIN refund_op_info infe on b.audit_op_id = infe.OP_ID

;

select * from REFUND_CHECK

select * from REFUND_OP_INFO

select * from refund_order_trans_inf
