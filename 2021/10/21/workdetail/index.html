pdf模板填充 outputStream 需要先建立文件夹

/**
 *
 * @author lh
 * @date 2021/10/28 14:40
 * @param Map  填充map
 * @param businessType  文件夹名称
 * @param pdfTemplate   pdf模板
 * @param file_prefix   文件路径前缀
 * @return java.lang.String
 */
public static String  createPDF(HashMap<String,String> Map, String businessType, String pdfTemplate,String file_prefix) {
    if (MapUtils.isEmpty(Map)) return null;
    PdfReader reader = null;
    AcroFields s = null;
    PdfStamper ps = null;
    ByteArrayOutputStream bos = null;
    String pdf_Name = UUID.randomUUID().toString();

    try {
            //赋值
            InputStream is = PdfUtil.class.getResourceAsStream("/" + pdfTemplate);
            reader = new PdfReader(is);
            bos = new ByteArrayOutputStream();
            ps = new PdfStamper(reader, bos);
            s = ps.getAcroFields();

            //指定中文字体
            BaseFont bold = BaseFont.createFont("/msyh.ttc,0", BaseFont.IDENTITY_H, BaseFont.EMBEDDED);
            if (bold != null)  s.addSubstitutionFont(bold);

            for (Map.Entry<String, String> entry : Map.entrySet()) {
                String key = entry.getKey();
                String value = entry.getValue();
                s.setField(key, value);
            }

            // 如果为false那么生成的PDF文件还能编辑，一定要设为true
            ps.setFormFlattening(true);
            ps.close();

            File file = new File(file_prefix + "/" + businessType);
            if (!file.exists()) file.mkdir();

            //outputStream 需要先建立文件夹
            FileOutputStream fos = new FileOutputStream(file_prefix+"/"+businessType+"/"+pdf_Name+ ".pdf");
            fos.write(bos.toByteArray());
            fos.close();

            is.close();
            bos.close();

    } catch (IOException | DocumentException e) {
        logger.error("createPdf-error",e);
    } finally {
        try {
            if (bos != null) {
                bos.close();
            }
            if (reader != null) {
                reader.close();
            }
            if (ps != null) {
                ps.close();
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
    return businessType + "/" + pdf_Name + ".pdf";
}

/**
 * @author lh
 * @date 2022-01-11 17:06
 */
public class serializableStreamDemo {

    //定义接口
    interface GetLambdaName extends Serializable {

        final static String METHOD_NAME = "writeReplace";

        //定义方法
        default String getLambdaMethodName() {

            String implMethodName = null;

            try {
                //获取当前对象的Class对象
                Class<? extends GetLambdaName> aClass = this.getClass();
                //获取 writeReplace 方法
                Method method = aClass.getDeclaredMethod(METHOD_NAME);
                //设置 访问权限
                method.setAccessible(true);
                //执行方法 获取  SerializedLambda
                SerializedLambda serializedLambda = (SerializedLambda) method.invoke(this);
                //获取方法
                implMethodName = serializedLambda.getImplMethodName();

            } catch (Exception e) {
                e.printStackTrace();
            }


            return implMethodName;
        }

    }


    //使用方法  接口可以多继承
    interface MyFunction<T, R> extends Function<T, R>, GetLambdaName {

    }

    public static void main(String[] args) {
        final MyFunction<User, String> getUsername = User::getUsername;
//        String lambdaMethodName = getUsername.getLambdaMethodName();
        String lambdaMethodName = "getUsername";

        System.out.println(lambdaMethodName);
        String get = Introspector.decapitalize(lambdaMethodName.replace("get", ""));
        System.out.println(get);
    }


}

springBoot 扩展机制 ——Spring.factories文件 如何将外部项目加入Spring容器
▶ 简介
简单说用于模块之间解耦，外部项目下各种配置、类加入到Spring容器中进行管理，类似JAVA中SPI机制。机制自行百度。
▶ 方式
将外部类引入IOC容器两种方式
1.@import
2.加入Spring.factories文件 — 主讲这个
3.SpringFactoriesLoader类会去扫描加载文件下定义类，加入容器

	org.springframework.boot.autoconfigure.EnableAutoConfiguration=\
  	com.mostsnails.common.base.utils.config.WebLogAutoConfiguration,\   //Configuration配置类中的bean
  	com.mostsnails.common.base.utils.config.RedisAutoConfiguration,\    //Configuration配置类中的bean
  	com.mostsnails.common.base.utils.test.TestClass                     //Component 修饰的bean
 
   

注意这里使用EnableAutoConfiguration ，SpringBoot启动类@SpringBootApplication注解包下有@EnableAutoConfiguration和@ComponentScan 具体作用开启自动装备，将满足条件@Configuration配置加入到SpringBoot中

@EnableAutoConfiguration中@Import(AutoConfigurationImportSelector.class)
AutoConfigurationImportSelector.class 加载 Spring.factories 中 bean 加载到容器中

AopContext

原来在springAOP的用法中，只有代理的类才会被切入，我们在controller层调用service的方法的时候，是可以被切入的，但是如果我们在service层 A方法中，调用B方法，切点切的是B方法，那么这时候是不会切入的，解决办法就是如上所示，在A方法中使用((Service)AopContext.currentProxy()).B() 来调用B方法，这样一来，就能切入了！

@EnableAspectJAutoProxy(exposeProxy = true)//开启spring注解aop配置的支持
public class SpringConfiguration {
}
